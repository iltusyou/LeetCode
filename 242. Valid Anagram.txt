/**
 * @param {string} s
 * @param {string} t
 * @return {boolean}
 */
var isAnagram = function(s, t) {
    
    var p = true;

    for (var i = 0; i < s.length; i++) {
        if (t.indexOf(s.charAt(i), 0) == -1) {
            p = false;
            break;
        }
        else {
            t=t.replace(s.charAt(i),"");
        }            
    }
    if (t != "") {
        p = false;
    }
    return p;
};
